2 Add Two Sum

Add Two Sum

这道题也是比较简单的，主要考察的就是链表的遍历。假设链表l1和l2的长度分别是m和n，则时间复杂度为o(m)+o(n)，注意其中第一个node不能要，所以return为l.Next，注意操作，并没有什么难度的。

思路

package solution

type ListNode struct {
    Val  int
    Next *ListNode
}

// add two num, eg: (2 -> 4 -> 3) + (5 -> 6 -> 4) -> (7 -> 0 -> 8)
func addTwoNumbers(l1 *ListNode, l2 *ListNode) *ListNode {

    // listNode for return
    l := &ListNode{}

    // to maintain the loop
    vL1, vL2, vL := l1, l2, l
    carry := 0
    for vL1 != nil || vL2 != nil {
        a, b, sum := 0, 0, 0
        if vL1 != nil {
            a = vL1.Val
            vL1 = vL1.Next
        }
        if vL2 != nil {
            b = vL2.Val
            vL2 = vL2.Next
        }
        sum = a + b +carry

        vL.Next = &ListNode{
            Val: sum % 10,
        }
        vL = vL.Next

        // update flag for next time
        carry = sum / 10
    }

    // check the last num
    if carry > 0 {
        vL.Next = &ListNode{
            Val: carry,
        }
    }
    return l.Next
}

测试

package solution

import (
    "log"
    "testing"
)

func TestAddTwoSum(t *testing.T) {

    // 所有测试用例
    cases := []struct {
        l1   *ListNode
        l2   *ListNode
        want []int
    }{
        {
            l1: &ListNode{
                Val: 2,
                Next: &ListNode{
                    Val: 4,
                    Next: &ListNode{
                        Val: 3,
                    },
                },
            },
            l2: &ListNode{
                Val: 5,
                Next: &ListNode{
                    Val: 6,
                    Next: &ListNode{
                        Val: 4,
                    },
                },
            },
            want: []int{7, 0, 8},
        },
        {
            l1: &ListNode{
                Val: 2,
                Next: &ListNode{
                    Val: 4,
                    Next: &ListNode{
                        Val: 6,
                    },
                },
            },
            l2: &ListNode{
                Val: 5,
                Next: &ListNode{
                    Val: 6,
                    Next: &ListNode{
                        Val: 4,
                    },
                },
            },
            want: []int{7, 0, 1, 1},
        },
    }

    // 遍历执行测试
    for i, c := range cases {

        // 获取返回链表，并转换为list，方便比较
        l := addTwoNumbers(c.l1, c.l2)
        ret := make([]int, 0)
        lPoint := l
        for lPoint != nil {
            ret = append(ret, lPoint.Val)
            lPoint = lPoint.Next
        }
        log.Println(ret)

        // 比较
        if len(ret) != len(c.want) {
            t.Errorf("no.%v case not pass, len(ret) != len(want)", i)
        }
        for j:=0; j<len(ret); j++ {
            if ret[j] != c.want[j] {
                t.Errorf("no.%v case not pass, ret[%v] != want[%v]", i, j, j)
            }
        }
    }
}
